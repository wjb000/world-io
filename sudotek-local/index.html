async function loadSelectedModel() { const modelSelect = document.getElementById('global-model'); const statusEl = document.getElementById('model-status'); const nameEl = document.getElementById('model-name'); if (!modelSelect || !statusEl || !nameEl) { console.error('Required elements not found - make sure IDs match'); return; } const modelId = modelSelect.value; statusEl.innerHTML = `Downloading...`; nameEl.innerHTML = modelId.split('-')[0]; try { engine = await webllm.CreateMLCEngine(modelId, { initProgressCallback: (progress) => { const pct = Math.round(progress.progress * 100); if (statusEl) statusEl.innerHTML = `${pct}%`; if (pct % 20 === 0) { logToPanel(`Downloading ${modelId}... ${pct}%`, 'system'); } } }); currentModel = modelId; isModelLoaded = true; if (statusEl) statusEl.innerHTML = `READY (cached)`; logToPanel(`✅ Model ${modelId} loaded successfully (now cached in browser)`, 'system'); showToast('Model downloaded and cached! Future loads will be instant.', false, 4000); } catch (e) { if (statusEl) statusEl.innerHTML = `FAILED`; showToast('Failed to load model: ' + e.message, true); } }
